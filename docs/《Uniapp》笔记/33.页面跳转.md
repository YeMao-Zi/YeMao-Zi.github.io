```js
export const scrollToWithId = (
  anchorId,
  targetId,
  proxy,
) => {
  uni
    .createSelectorQuery()
    .select(anchorId)
    .boundingClientRect((anchor) => {
      uni
        .createSelectorQuery()
        // #ifdef MP-WEIXIN
        .in(proxy)
        // #endif
        .select(targetId)
        .boundingClientRect((rect) => {
          const newAnchor = anchor as UniApp.NodeInfo;
          const newRect = rect as UniApp.NodeInfo;
          uni.pageScrollTo({
            scrollTop: Math.abs((newRect?.top ?? 0) - (newAnchor?.top ?? 0)),
            duration: 300,
          });
        })
        .exec();
    })
    .exec();
};
```
