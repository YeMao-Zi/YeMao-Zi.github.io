forwardRefе’ҢuseImperativeHandle

# forwardRef

React.forwardRefеӯ—йқўж„ҸжҖқзҗҶи§ЈдёәиҪ¬еҸ‘RefпјҢе®ғдјҡеҲӣе»әдёҖдёӘReactз»„д»¶пјҢиҝҷдёӘз»„д»¶иғҪеӨҹе°Ҷе…¶жҺҘеҸ—зҡ„ ref еұһжҖ§иҪ¬еҸ‘еҲ°е…¶з»„д»¶ж ‘дёӢзҡ„еҸҰдёҖдёӘз»„д»¶дёӯгҖӮе…¶дё»иҰҒдҪңз”ЁжҳҜпјҡ

# 1.иҪ¬еҸ‘refsеҲ°DOMз»„д»¶

ref дёҚеғҸ props дҪңдёәеҸӮж•°еҸҜд»Ҙдј йҖ’пјҢжүҖд»ҘиҰҒжғідј йҖ’ ref еҫ—з”Ё forwardRefгҖӮ

const FancyButton = React.forwardRef((props, ref) => (
  <button ref={ref} className="FancyButton">
    {props.children}
  </button>
));

// You can now get a ref directly to the DOM button:
const ref = useRef(null);

const Father=()=>{
    const handleClick=()=>{
        const RefValue = ref.current
        RefValue.handleCancel() // еҸҜд»ҘзӣҙжҺҘдҪҝз”Ёиў«иҪ¬еҸ‘зҡ„ж–№жі•
    }
    return (
        <button onClick={handleClick}>
        <FancyButton ref={ref}>Click me!</FancyButton>;
    )
}

еҰӮжһңдёҖдёӘз»„д»¶иў«еӨҡж¬ЎдҪҝз”ЁпјҢжӯЈеёёжғ…еҶөдёӢжғіиҰҒи°ғз”Ёе…¶з»„д»¶еҶ…зҡ„ж–№жі•йңҖиҰҒдј е…ҘpropsжқҘи°ғз”ЁпјҢжҜҸж¬Ўдј е…Ҙзҡ„иҜқе°ұжҜ”иҫғеӨҡдҪҷгҖӮжүҖд»Ҙе°ұеј•е…ҘдәҶReact.forwardRefгҖӮпјҲеңЁtsxдёӯзҡ„иҝҗз”ЁпјүгҖӮ

interface SelectFileModalRef {
  handleShowModal: () => void;
  handleCancel: () => void;
}
const SelectFileModal = forwardRef<SelectFileModalRef, Props>(
  (props: Props, ref: Ref<SelectFileModalRef>) => {
   useImperativeHandle(ref, () => ({
      handleShowModal,
      handleCancel,
    }));
  }
);

# 2.еңЁй«ҳйҳ¶з»„д»¶дёӯиҪ¬еҸ‘ refs

йҰ–е…ҲиҰҒжҗһжё…жҘҡд»Җд№ҲжҳҜй«ҳйҳ¶з»„д»¶пјҡй«ҳйҳ¶з»„д»¶пјҲHOCпјүе®ғжҳҜдёҖз§ҚеҹәдәҺ React зҡ„з»„еҗҲзү№жҖ§иҖҢеҪўжҲҗзҡ„и®ҫи®ЎжЁЎејҸгҖӮ

е…·дҪ“иҖҢиЁҖпјҢй«ҳйҳ¶з»„д»¶жҳҜеҸӮж•°дёәз»„д»¶пјҢиҝ”еӣһеҖјдёәж–°з»„д»¶зҡ„еҮҪж•°

// жӯӨеҮҪж•°жҺҘж”¶дёҖдёӘз»„д»¶...
function withSubscription(WrappedComponent, selectData) {
  // ...е№¶иҝ”еӣһеҸҰдёҖдёӘз»„д»¶...
  return class extends React.Component {
    constructor(props) {
      super(props);
      this.handleChange = this.handleChange.bind(this);
      this.state = {
        data: selectData(DataSource, props)
      };
    }

    componentDidMount() {
      // ...иҙҹиҙЈи®ўйҳ…зӣёе…ізҡ„ж“ҚдҪң...
      DataSource.addChangeListener(this.handleChange);
    }

    componentWillUnmount() {
      DataSource.removeChangeListener(this.handleChange);
    }

    handleChange() {
      this.setState({
        data: selectData(DataSource, this.props)
      });
    }

    render() {
      // ... е№¶дҪҝз”Ёж–°ж•°жҚ®жёІжҹ“иў«еҢ…иЈ…зҡ„з»„д»¶!
      // иҜ·жіЁж„ҸпјҢжҲ‘д»¬еҸҜиғҪиҝҳдјҡдј йҖ’е…¶д»–еұһжҖ§
      return <WrappedComponent data={this.state.data} {...this.props} />;
    }
  };
}

function logProps(Component) {
  class LogProps extends React.Component {
    componentDidUpdate(prevProps) {
      console.log('old props:', prevProps);
      console.log('new props:', this.props);
    }

    render() {
      const {forwardedRef, ...rest} = this.props;

      // е°ҶиҮӘе®ҡд№үзҡ„ prop еұһжҖ§ вҖңforwardedRefвҖқ е®ҡд№үдёә ref
      return <Component ref={forwardedRef} {...rest} />;
    }
  }

  // жіЁж„Ҹ React.forwardRef еӣһи°ғзҡ„з¬¬дәҢдёӘеҸӮж•° вҖңrefвҖқгҖӮ
  // жҲ‘д»¬еҸҜд»Ҙе°Ҷе…¶дҪңдёәеёёи§„ prop еұһжҖ§дј йҖ’з»ҷ LogPropsпјҢдҫӢеҰӮ вҖңforwardedRefвҖқ
  // з„¶еҗҺе®ғе°ұеҸҜд»Ҙиў«жҢӮиҪҪеҲ°иў« LogProps еҢ…иЈ№зҡ„еӯҗз»„д»¶дёҠгҖӮ
  return React.forwardRef((props, ref) => {
    return <LogProps {...props} forwardedRef={ref} />;
  });
}
